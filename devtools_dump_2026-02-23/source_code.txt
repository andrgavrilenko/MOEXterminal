function loadMacro(){
  var sel=document.getElementById('macro-sel');
  var id=sel?sel.value:'';
  if(!id) return;
  if(macroChartData&&macroChartData._id===id){
    setTimeout(function(){drawMacro(macroChartData);},50);
    return;
  }
  fetch('/api/macro/'+id)
    .then(function(r){return r.json()})
    .then(function(data){macroChartData=data;macroChartData._id=id;setTimeout(function(){drawMacro(data);},50);})
    .catch(function(e){console.error('macro load error',e);});
}

function drawMacro(data){

  if(!dates.length) return;
  if(macroRange!=='all'){

    var cutoff=new Date(new Date().getFullYear()-yrs,0,1).toISOString().slice(0,10);
    var f=[];var cl=dates[0].length;var cc=cutoff.slice(0,cl);for(var i=0;i<dates.length;i++){if(dates[i]>=cc)f.push({d:dates[i],v:values[i]});}
    if(f.length){dates=f.map(x=>x.d);values=f.map(x=>x.v);}
  }
  var sel=document.getElementById('macro-sel');

  var cat=(lastData&&lastData.macro_catalog)||[];
  var item=cat.find(c=>c.id===macroChartData._id)||{};
  var trace={x:dates,y:values,name:name};

  // Берём РЕАЛЬНУЮ ширину видимого контейнера

  Plotly.react('macro-chart',[trace],{

      if(npTickers.length===1)p.set("ticker",npTickers[0]);
      if(npTag&&npTag!=="all")p.set("tag",npTag);
      r=await(await fetch(NP+"/feed?"+p)).json();
      var newItems=r.items||[];
      newItems.sort(function(a,b){return new Date(b.date)-new Date(a.date);});
      if(reset)npItems=newItems;else npItems=npItems.concat(newItems);
      npMore=newItems.length===50;npOff+=newItems.length;
    }
    npRender();
  }catch(e){var f=document.getElementById("npFeed");if(f)f.innerHTML='<div style="text-align:center;padding:20px;color:var(--td);font-size:10px">ERROR</div>';}
  npBusy=false;
}

function npRender(){
  var f=document.getElementById("npFeed");
  if(!f)return;
  if(npItems.length===0){f.innerHTML='<div style="text-align:center;padding:20px;color:var(--td);font-size:10px">NO NEWS</div>';return;}
  var h="";
  for(var i=0;i<npItems.length;i++){
    var it=npItems[i];
    var brk=(it.tags||[]).indexOf("breaking")>=0;
    var tks="";
    if(it.tickers){for(var j=0;j<it.tickers.length;j++){tks+='<span class="ntk" onclick="event.stopPropagation();window._npAdd(\''+it.tickers[j]+'\')">'+it.tickers[j]+'</span>';}}
    var tgs="";
    if(it.tags){for(var j=0;j<it.tags.length;j++){tgs+='<span class="ntg-'+it.tags[j]+'">'+it.tags[j].toUpperCase()+'</span>';}}
    h+='<div class="ni'+(brk?' nbrk':'')+'" onclick="this.classList.toggle(\'nexp\')">'
      +'<div class="ntm">'+npFmt(it.date)+'</div>'
      +'<div class="nnb">'
      +(tks?'<div class="nnm">'+tks+'</div>':'')
      +'<div class="ntxt">'+npEsc(it.text)+'</div>'
      +(tgs?'<div class="ntg">'+tgs+'</div>':'')
      +'</div></div>';
  }
  f.innerHTML=h;
}

function npAddTicker(tk){
  tk=tk.toUpperCase().trim();
  if(!tk||npTickers.indexOf(tk)>=0)return;
  npTickers.push(tk);
  document.getElementById("npTicker").value="";
  renderFilters();npLoad(true);
}
window._npAdd=npAddTicker;
